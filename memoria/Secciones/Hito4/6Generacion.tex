        recolecta_subs(Decs)
        gen_cod(Insts)

        while not es_vacia(sub_pendientes)

    gen_paso_PFml(proc(id,PFml,Decs,Is),PReal):

        if PFml = pref(T,id) v not es_designador(PReal) then
            emit desapila_ind()
        else
